CREATE TABLE Users(
	ID INT IDENTITY PRIMARY KEY,
	[Name] NVARCHAR(31),
	Surname NVARCHAR(31),
	Username VARCHAR(31) UNIQUE,
	[Password] NVARCHAR(512) NOT NULL
)

CREATE TABLE Blogs(
	ID INT IDENTITY PRIMARY KEY,
	Title NVARCHAR(31),
	Describtion NVARCHAR(63),
	[User_ID] INT FOREIGN KEY REFERENCES Users(ID)
)
